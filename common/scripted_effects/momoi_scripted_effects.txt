KOM_momoi_left_ready_change = {
	add_to_variable = {KOM_left_ready = KOM_ready_temp}
	if = {
		limit = {
			check_variable = { KOM_ready_temp < 0 }
		}
		multiply_temp_variable = { KOM_ready_temp = -1 }
		custom_effect_tooltip = KOM_left_ready_decrease_tt
	}
	else = {
		custom_effect_tooltip = KOM_left_ready_increase_tt
	}
	clamp_variable = {
		var = KOM_left_ready
		min = 0
		max = 100
	}
}
